fn insert_when(&mut self, elt: T, f: |&T, &T| -> bool)

Insert elt before the first x in the list where f(x, elt) is true, or at the end.

O(N)

Example

fn main() { use std::collections::DList; let mut a: DList<int> = DList::new(); a.push(2i); a.push(4); a.push(7); a.push(8); // insert 11 before the first odd number in the list a.insert_when(11, |&e, _| e % 2 == 1); for e in a.iter() { println!("{}", e); // prints 2, then 4, then 11, then 7, then 8 } }
use std::collections::DList;

let mut a: DList<int> = DList::new();

a.push(8);

}
