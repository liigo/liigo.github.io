std::macros::format

macro_rules! format(
    ($($arg:tt)*) => (
        format_args!(::std::fmt::format, $($arg)*)
    )
)

Use the syntax described in std::fmt to create a value of type ~str. See std::fmt for more information.

Example

format!("test");
format!("hello {}", "world!");
format!("x = {}, y = {y}", 10, y = 30);
