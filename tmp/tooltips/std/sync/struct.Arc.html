An atomically reference counted wrapper for shared state.

Example

In this example, a large vector of floats is shared between several tasks. With simple pipes, without Arc, a copy would have to be made for each task.

use std::sync::Arc; use std::thread::Thread; fn main() { let numbers = Vec::from_fn(100, |i| i as f32); let shared_numbers = Arc::new(numbers); for _ in range(0u, 10) { let child_numbers = shared_numbers.clone(); Thread::spawn(move || { let local_numbers = child_numbers.as_slice(); // Work with the local numbers }).detach(); } }
use std::sync::Arc;
use std::thread::Thread;

fn main() {
    let numbers = Vec::from_fn(100, |i| i as f32);
    let shared_numbers = Arc::new(numbers);

    for _ in range(0u, 10) {
        let child_numbers = shared_numbers.clone();

        Thread::spawn(move || {
            let local_numbers = child_numbers.as_slice();

            // Work with the local numbers
        }).detach();
    }
}
