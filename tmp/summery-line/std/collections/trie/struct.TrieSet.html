fn each_reverse(&self, f: |&uint| -> bool) -> bool

Visits all values in reverse order. Aborts traversal when f returns false. Returns true if f returns true for all elements.

Example

fn main() { use std::collections::TrieSet; let set: TrieSet = [1, 2, 3, 4, 5].iter().map(|&x| x).collect(); let mut vec = Vec::new(); assert_eq!(true, set.each_reverse(|&x| { vec.push(x); true })); assert_eq!(vec, vec![5, 4, 3, 2, 1]); // Stop when we reach 3 let mut vec = Vec::new(); assert_eq!(false, set.each_reverse(|&x| { vec.push(x); x != 3 })); assert_eq!(vec, vec![5, 4, 3]); }
use std::collections::TrieSet;

let set: TrieSet = [1, 2, 3, 4, 5].iter().map(|&x| x).collect();

let mut vec = Vec::new();
assert_eq!(true, set.each_reverse(|&x| { vec.push(x); true }));
assert_eq!(vec, vec![5, 4, 3, 2, 1]);

// Stop when we reach 3
let mut vec = Vec::new();
assert_eq!(false, set.each_reverse(|&x| { vec.push(x); x != 3 }));
assert_eq!(vec, vec![5, 4, 3]);
