pub trait MutableMap<K, V>: Mutable + Map<K, V> {
    fn swap(&mut self, k: K, v: V) -> Option<V>;
    fn pop(&mut self, k: &K) -> Option<V>;
    fn find_mut(&'a mut self, key: &K) -> Option<&'a mut V>;

    fn insert(&mut self, key: K, value: V) -> bool { ... }
    fn remove(&mut self, key: &K) -> bool { ... }
}

A key-value store (map) where the values can be modified.

Required Methods

fn swap(&mut self, k: K, v: V) -> Option<V>

Inserts a key-value pair into the map. If the key already had a value present in the map, that value is returned. Otherwise, None is returned.

Example

fn main() { use std::collections::HashMap; let mut map = HashMap::new(); assert_eq!(map.swap("a", 37i), None); assert_eq!(map.is_empty(), false); map.insert("a", 1i); assert_eq!(map.swap("a", 37i), Some(1i)); assert_eq!(map.get(&"a"), &37i); }
use std::collections::HashMap;

let mut map = HashMap::new();
assert_eq!(map.swap("a", 37i), None);
assert_eq!(map.is_empty(), false);

map.insert("a", 1i);
assert_eq!(map.swap("a", 37i), Some(1i));
assert_eq!(map.get(&"a"), &37i);

fn pop(&mut self, k: &K) -> Option<V>

Removes a key from the map, returning the value at the key if the key was previously in the map.

Example

fn main() { use std::collections::HashMap; let mut map: HashMap<&str, int> = HashMap::new(); map.insert("a", 1i); assert_eq!(map.pop(&"a"), Some(1i)); assert_eq!(map.pop(&"a"), None); }
use std::collections::HashMap;

let mut map: HashMap<&str, int> = HashMap::new();
map.insert("a", 1i);
assert_eq!(map.pop(&"a"), Some(1i));
assert_eq!(map.pop(&"a"), None);

fn find_mut(&'a mut self, key: &K) -> Option<&'a mut V>

Returns a mutable reference to the value corresponding to the key.

Example

fn main() { use std::collections::HashMap; let mut map = HashMap::new(); map.insert("a", 1i); match map.find_mut(&"a") { Some(x) => *x = 7i, None => (), } assert_eq!(map.get(&"a"), &7i); }
use std::collections::HashMap;

let mut map = HashMap::new();
map.insert("a", 1i);
match map.find_mut(&"a") {
    Some(x) => *x = 7i,
    None => (),
}
assert_eq!(map.get(&"a"), &7i);

Provided Methods

fn insert(&mut self, key: K, value: V) -> bool

Inserts a key-value pair into the map. An existing value for a key is replaced by the new value. Returns true if the key did not already exist in the map.

Example

fn main() { use std::collections::HashMap; let mut map = HashMap::new(); assert_eq!(map.insert("key", 2i), true); assert_eq!(map.insert("key", 9i), false); assert_eq!(map.get(&"key"), &9i); }
use std::collections::HashMap;

let mut map = HashMap::new();
assert_eq!(map.insert("key", 2i), true);
assert_eq!(map.insert("key", 9i), false);
assert_eq!(map.get(&"key"), &9i);

fn remove(&mut self, key: &K) -> bool

Removes a key-value pair from the map. Returns true if the key was present in the map.

Example

fn main() { use std::collections::HashMap; let mut map = HashMap::new(); assert_eq!(map.remove(&"key"), false); map.insert("key", 2i); assert_eq!(map.remove(&"key"), true); }
use std::collections::HashMap;

let mut map = HashMap::new();
assert_eq!(map.remove(&"key"), false);
map.insert("key", 2i);
assert_eq!(map.remove(&"key"), true);
