use std::os;

let key = "PATH";
match os::getenv_as_bytes(key) {
    Some(paths) => {
        for path in os::split_paths(paths).iter() {
            println!("'{}'", path.display());
        }
    }
    None => println!("{} is not defined in the environment.", key)
}
