Reads from a fixed-size byte slice

Example

fn main() { #![allow(unused_must_use)] use std::io::BufReader; let mut buf = [0, 1, 2, 3]; let mut r = BufReader::new(buf); assert_eq!(r.read_to_end().unwrap(), vec!(0, 1, 2, 3)); }
use std::io::BufReader;

let mut buf = [0, 1, 2, 3];
let mut r = BufReader::new(buf);
